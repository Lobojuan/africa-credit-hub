app.post("/api/telco/score", async (req, res) => {
  try {
    const { accountId } = req.body;
    if (!accountId) return res.status(400).json({ error: "accountId required" });

    // Mocked Aggregator Data
    const rawTransactions = [
        { type: "CREDIT", amount: 950, category: "BUSINESS", narration: "merchant sale" },
        { type: "DEBIT", amount: 50, category: "UTILITY", narration: "electricity bill" },
        { type: "DEBIT", amount: 10, category: "TELCO", narration: "airtime advance" }
    ];

    const inflows = rawTransactions.filter(t => t.type === "CREDIT").reduce((sum, t) => sum + t.amount, 0);
    const utilityPayments = rawTransactions.filter(t => t.category === "UTILITY").length;
    const emergencyLoans = rawTransactions.filter(t => t.narration.includes("advance")).length;
    const kpiProfile = JSON.stringify({ inflows_usd: inflows, utility_payments: utilityPayments, airtime_advances: emergencyLoans });

    // STEP 1: GPT-4o (Quant Math)
    const gptPrompt = `Analyze this telecom profile: ${kpiProfile}. Calculate risk score (1-5, 5 is highest risk) based on cash flow and liquidity stress. Calculate credit limit (max 20% of inflows). Output JSON ONLY: {"risk_score": number, "credit_limit_usd": number}`;
    const gptCompletion = await openai.chat.completions.create({ model: "gpt-4o", response_format: { type: "json_object" }, messages: [{ role: "user", content: gptPrompt }], temperature: 0.1 });
    const quantData = JSON.parse(gptCompletion.choices[0].message.content || "{}");

    // STEP 2: Claude 3.5 (Compliance Rationale)
    const claudePrompt = `You are a compliance officer. The quantitative AI assigned this profile a risk score of ${quantData.risk_score} and a limit of $${quantData.credit_limit_usd}. Raw data: ${kpiProfile}. Write a regulatory-compliant 'reason_code' and a 'detailed_rationale' explaining the decision. Determine 'approval_recommendation' (true if risk <= 3). Output JSON ONLY: {"reason_code": "string", "detailed_rationale": "string", "approval_recommendation": boolean}`;
    const claudeResponse = await anthropic.messages.create({ model: "claude-3-5-sonnet-20241022", max_tokens: 500, messages: [{ role: "user", content: claudePrompt }] });
    const claudeText = claudeResponse.content[0].type === 'text' ? claudeResponse.content[0].text : "{}";
    
    res.json({ ...quantData, ...JSON.parse(claudeText), kpi_used: JSON.parse(kpiProfile) });
  } catch (error: any) {
    res.status(500).json({ error: error.message });
  }
});

app.post("/api/chat", async (req, res) => {
  try {
    const { message, contextData } = req.body;
    
    // STEP 1: The Brain (GPT-4o) extracts data & performs logic
    const gptDataPrompt = `User query: "${message}". Context: ${JSON.stringify(contextData)}. Extract the core financial facts, perform any requested calculations, and output a raw bulleted list of facts to answer the query. Do not format for the user, just provide raw data.`;
    
    const gptCompletion = await openai.chat.completions.create({
      model: "gpt-4o",
      messages: [{ role: "user", content: gptDataPrompt }],
      temperature: 0.1
    });
    const rawFacts = gptCompletion.choices[0].message.content;

    // STEP 2: The Voice (Claude 3.5) formats the response with high EQ
    const claudeVoicePrompt = `You are the Africa Credit Hub AI Assistant. A user asked: "${message}". 
    Here are the raw mathematical facts retrieved by our data engine:
    ${rawFacts}
    
    Draft a highly professional, empathetic, and regulatory-compliant response to the user using ONLY these facts. Format beautifully with markdown if needed.`;

    const claudeResponse = await anthropic.messages.create({
      model: "claude-3-5-sonnet-20241022",
      max_tokens: 1000,
      messages: [{ role: "user", content: claudeVoicePrompt }]
    });

    const finalResponse = claudeResponse.content[0].type === 'text' ? claudeResponse.content[0].text : "Sorry, I could not process that request.";
    
    res.json({ text: finalResponse });
  } catch (error: any) {
    res.status(500).json({ error: error.message });
  }
});
